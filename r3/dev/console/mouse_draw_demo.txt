| Console Mouse Drawing Demo
| Cross-platform drawing application with mouse support
| Works with both Linux and Windows console libraries

| Uncomment the appropriate library for your platform:
| ^r3/lib/linux-console.r3  | For Linux
| ^r3/lib/win-console.r3    | For Windows

|------- Canvas State -------
#canvas * 8192  | 128x64 canvas buffer (1 byte per cell)
#canvas-width 120
#canvas-height 40

|------- Drawing State -------
#mouse-x 0
#mouse-y 0
#mouse-btn 0
#prev-btn 0
#current-brush 1
#current-color 1
#drawing-mode 0  | 0=draw, 1=erase

|------- Brush Definitions -------
| Brush characters
#brushes ( 
    $2588  | 0: Full block █
    $2593  | 1: Dark shade ▓
    $2592  | 2: Medium shade ▒
    $2591  | 3: Light shade ░
    $2022  | 4: Bullet •
    $25CF  | 5: Circle ●
    $2B1B  | 6: Black square ⬛
    $2B1C  | 7: White square ⬜
    $2605  | 8: Star ★
    $2764  | 9: Heart ❤
)

#brush-count 10

|------- Color Palette -------
:set-draw-color | color --
    1 =? ( .Redl ; )
    2 =? ( .Greenl ; )
    3 =? ( .Bluel ; )
    4 =? ( .Yellowl ; )
    5 =? ( .Magental ; )
    6 =? ( .Cyanl ; )
    7 =? ( .Whitel ; )
    drop .Whitel ;

|------- Canvas Functions -------
:canvas-clear | --
    'canvas 0 canvas-width canvas-height * cfill ;

:canvas-at | x y -- addr
    canvas-width * + 'canvas + ;

:canvas-get | x y -- char
    canvas-at c@ ;

:canvas-set | char x y --
    canvas-at c! ;

:canvas-draw | --
    2 2 .at
    0 ( canvas-height <?
        dup >r
        0 ( canvas-width <?
            dup over canvas-get
            dup 0? ( 2drop " " .write ; )
            
            | Get color from high nibble
            dup 4 >> set-draw-color
            
            | Get brush from low nibble
            $0F and 3 << 'brushes + @ .emit
            
            1 +
        ) drop
        r> 1 +
        .cr
    ) drop
    .Reset ;

|------- UI Functions -------
:draw-frame | --
    .home .Cyanl .Bold
    0 0 .at "┌" .write
    1 ( cols 1 - <? "─" .write 1 + ) drop
    "┐" .write
    
    1 ( rows 1 - <?
        dup 0 .at "│" .write
        dup cols 1 - .at "│" .write
        1 + ) drop
    
    0 rows 1 - .at "└" .write
    1 ( cols 1 - <? "─" .write 1 + ) drop
    "┘" .write
    .Reset ;

:draw-toolbar | --
    2 canvas-height 3 + .at
    .Yellow "┤ " .write
    .Yellowl .Bold "Brush: " .write
    .Whitel current-brush "%d " .print
    
    .Yellow "│ " .write
    .Yellowl .Bold "Color: " .write
    current-color set-draw-color "█ " .write
    
    .Yellow "│ " .write
    .Yellowl .Bold "Mode: " .write
    .Whitel drawing-mode 0? ( drop "DRAW" ; ) drop "ERASE"
    .write
    
    .Yellow " │" .write
    .Reset
    
    2 canvas-height 4 + .at
    .Cyan
    "[1-9] Brush  [C] Color  [M] Mode  [SPACE] Clear  [ESC] Exit" .write
    .Reset ;

:draw-help | --
    2 canvas-height 5 + .at
    .Magenta
    "Left Click: Draw  |  Right Click: Erase  |  Drag to draw continuous" .write
    .Reset ;

:draw-mouse-cursor | --
    mouse-x mouse-y .at
    mouse-btn 1 and 1? ( .Redl "+" ; ) drop
    mouse-btn 2 and 1? ( .Bluel "×" ; ) drop
    .Greenl "◆"
    .write .Reset ;

:draw-screen | --
    .cls
    draw-frame
    canvas-draw
    draw-toolbar
    draw-help
    draw-mouse-cursor ;

|------- Mouse Handling -------
:process-mouse | --
    | Get mouse position (adjust for canvas offset)
    evtmxy 2 - swap 2 - swap
    
    | Clamp to canvas bounds
    0 max canvas-height 1 - min 'mouse-y !
    0 max canvas-width 1 - min 'mouse-x !
    
    | Get button state
    evtmb 'mouse-btn !
    
    | Left button - draw
    mouse-btn 1 and 1? (
        drawing-mode 0? (
            | Draw mode
            current-color 4 << current-brush or
            mouse-x mouse-y canvas-set
            draw-screen
        ) drop
        
        | Check if entering draw mode
        prev-btn 1 and 0? ( draw-screen ) drop
    ) drop
    
    | Right button - erase
    mouse-btn 2 and 1? (
        0 mouse-x mouse-y canvas-set
        draw-screen
        
        | Check if entering erase mode
        prev-btn 2 and 0? ( draw-screen ) drop
    ) drop
    
    | Update previous button state
    mouse-btn 'prev-btn !
    
    | Redraw cursor if mouse moved
    draw-mouse-cursor ;

|------- Keyboard Handling -------
:handle-key | key --
    | ESC - exit
    $1B =? ( exit ; )
    
    | Numbers 1-9 - select brush
    dup $31 >=? ( 
        dup $39 <=? (
            $30 - 1 - 
            dup 0 >=? (
                dup brush-count <? (
                    'current-brush ! 
                    draw-screen ;
                ) drop
            ) drop
        ) drop
    ) drop
    
    | C/c - cycle color
    dup $43 =? ( drop 1 ) swap
    $63 =? (
        current-color 1 + 8 mod 
        dup 0? ( drop 1 )
        'current-color !
        draw-screen ;
    ) drop
    
    | M/m - toggle mode
    dup $4D =? ( drop 1 ) swap
    $6D =? (
        drawing-mode 1 xor 'drawing-mode !
        draw-screen ;
    ) drop
    
    | SPACE - clear canvas
    $20 =? (
        canvas-clear
        draw-screen ;
    )
    
    drop ;

|------- Event Loop -------
#running 1

:exit
    0 'running ! ;

:main-loop | --
    ( running 1?
        | Check for resize
        .checksize 1? ( drop draw-screen )
        
        | Check for events
        inevt
        dup 1 =? ( drop evtkey handle-key )  | Keyboard
        dup 2 =? ( drop process-mouse )      | Mouse
        dup 4 =? ( drop draw-screen )        | Resize
        drop
        
        | Small delay
        10 ms
    ) drop ;

|------- Resize Handler -------
:on-resize-event
    | Recalculate canvas dimensions if needed
    cols 4 - 'canvas-width !
    rows 8 - 'canvas-height !
    canvas-width 0 max 120 min 'canvas-width !
    canvas-height 0 max 40 min 'canvas-height !
    draw-screen ;

|------- Main Program -------
:main
    | Initialize console
    init-console
    
    | Enable mouse support
    .enable-mouse
    
    | Set resize callback
    'on-resize-event .onresize
    
    | Initialize canvas
    canvas-clear
    
    | Initial state
    1 'current-brush !
    1 'current-color !
    0 'drawing-mode !
    
    | Hide cursor for cleaner display
    .hidec
    
    | Initial draw
    draw-screen
    
    | Welcome message
    2 canvas-height 6 + .at
    .Greenl .Bold
    "Welcome! Use your mouse to draw. Try different brushes and colors!" .write
    .Reset
    
    | Run main loop
    main-loop
    
    | Cleanup
    .disable-mouse
    .showc
    .cls
    .home
    .Reset
    "Thanks for drawing!" .println ;

| Program entry point
: main ;
